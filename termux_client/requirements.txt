# On Termux/Android, install via ONE of the following methods:
#
# Quick Fix (recommended — jika aioquic sudah terinstall tapi tidak ditemukan):
#   python -m pip install --upgrade --break-system-packages aioquic
#
#   Penjelasan: pip biasa di Termux terkadang menginstall ke lokasi berbeda
#   dari Python sistem. Flag --break-system-packages memaksa pip menulis
#   langsung ke system site-packages agar dapat diakses oleh semua Python.
#
# Method 1 (TUR pre-built binary — tanpa compile Rust):
#   pkg install tur-repo -y
#   pkg install python-cryptography -y
#   python -m pip install --upgrade --break-system-packages aioquic
#
# Method 2 (compile from source — slower, ~10-20 menit):
#   pkg install rust binutils clang make libffi openssl -y
#   export CARGO_BUILD_TARGET="$(rustc -Vv | grep 'host' | awk '{print $2}')"
#   export CFLAGS="-Wno-register" CXXFLAGS="-Wno-register"
#   python -m pip install --break-system-packages aioquic
#
# After installation, verify with:
#   python -c "import aioquic; print('OK:', aioquic.__version__)"
#
